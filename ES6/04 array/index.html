• 동일한 값으로 구성된 배열 두개더라도 서로 다른 배열이다(Imutable) 
              //a와 b는 다른 배열이다
          let a=[1,2,3]
          let b=[1,2,3]
  • 배열은 참조형이다 
              //a와 b는 같은배열
          let a = [1,2,3]
          let b = a
  • 배열 선언방법 
              let a = []
          let a = new Array(3)
  • 배열의 인덱스는 0부터 시작, 갯수 -1 으로 끝 
              //let arr = [1,2,3]
          arr[0] = 1
          arr[2] = 3
  • 배열의 값을 추가, 제거 
              arr.push() //배열 뒤 추가 
          arr.unshift() // 배열 앞 추가
          arr.pop() // 배열 뒤 제거
          arr.shift() // 배열 앞 제거
  • splice(배열의 값을 추가, 제거) 
              arr = [1,2,3]
          arr.splice(1,2) //1번쨰부터 2개 지움
          => arr = [1]
          arr.splice(2,1,4) //2번쨰부터 1개를 지우고 4 추가
           => arr = [1,2,4]
  • split , 글자를 이용해 배열생성 
              let str = 123456
          let arr = str.split('')
          arr = [1,2,3,4,5,6]
